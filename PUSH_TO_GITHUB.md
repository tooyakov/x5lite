# Команды для публикации в GitHub

# 1. Настройте git user (одноразово)
git config --global user.email "veagledesign@gmail.com"
git config --global user.name "tooyakov"

# 2. Сделайте commit
git commit -m "Initial commit: iOS app with WebView and Apple Pay"

# 3. Запушьте в GitHub
git push -u origin main
